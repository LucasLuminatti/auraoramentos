```sql
-- Retorna clientes que fazem aniversário em today+5d (cross-year safe)

  cliente_id UUID,
  cliente_nome TEXT,
  data_nascimento DATE,
  colab_email TEXT,
  ano_referencia INT

-- Lista emails de todos admins (has_role(admin) dinâmico, não hardcode)
buscar_admins_emails()
RETURNS TABLE (email TEXT)

-- Idempotência atomic — UNIQUE(cliente_id, ano_referencia)
aniversario_envios (
  cliente_id UUID,
  ano_referencia INT,
  status TEXT, -- 'sent' | 'failed' | 'skipped_no_owner'
  enviado_em TIMESTAMPTZ,
  resend_id TEXT,
  erro TEXT
)
```
