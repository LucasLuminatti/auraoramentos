```ts
type SelectKind = 'arquiteto' | 'none' | 'all';
// 'arquiteto' = arquiteto escolhido (objeto preenchido)
// 'none'      = "Nenhum arquiteto" (arquiteto IS NULL) — sempre disponível
// 'all'       = "Todos" — só aparece em mode='filter'

interface ArquitetoAutocompleteProps {
  value: string;
  onSelect: (arquiteto: ArquitetoOption | null, kind?: SelectKind) => void;
  placeholder?: string;
  className?: string;
  mode?: 'select' | 'filter';  // default 'select'
}
```

Em mode='select', `kind` é omitido (callers atuais ignoram → backwards-compatible).
Em mode='filter', `kind` é sempre passado ('arquiteto' | 'none' | 'all').
