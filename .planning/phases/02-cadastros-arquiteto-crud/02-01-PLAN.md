    /**
     * Aplica máscara de CPF: 000.000.000-00
     * Aceita string parcial; é idempotente (re-aplicar não duplica).
     */
    export function formatCPF(value: string): string {
      const digits = value.replace(/\D/g, "").slice(0, 11);
      if (digits.length <= 3) return digits;
      if (digits.length <= 6) return `${digits.slice(0, 3)}.${digits.slice(3)}`;
      if (digits.length <= 9) return `${digits.slice(0, 3)}.${digits.slice(3, 6)}.${digits.slice(6)}`;
      return `${digits.slice(0, 3)}.${digits.slice(3, 6)}.${digits.slice(6, 9)}-${digits.slice(9, 11)}`;
    }

    /**
     * Aplica máscara de telefone celular BR: (00) 00000-0000
     * Aceita string parcial; suporta apenas celular 11 dígitos (D-10).
     */
    export function formatTelefone(value: string): string {
      const digits = value.replace(/\D/g, "").slice(0, 11);
      if (digits.length === 0) return "";
      if (digits.length <= 2) return `(${digits}`;
      if (digits.length <= 7) return `(${digits.slice(0, 2)}) ${digits.slice(2)}`;
      return `(${digits.slice(0, 2)}) ${digits.slice(2, 7)}-${digits.slice(7, 11)}`;
    }

    /**
     * Auto-detecta CPF (11 dígitos) vs CNPJ (>=12 dígitos) e aplica máscara.
     * 000.000.000-00 ou 00.000.000/0000-00.
     */
    export function formatCpfCnpj(value: string): string {
      const digits = value.replace(/\D/g, "").slice(0, 14);
      if (digits.length <= 11) return formatCPF(digits);
      // CNPJ progressivo
      if (digits.length <= 12) return `${digits.slice(0, 2)}.${digits.slice(2, 5)}.${digits.slice(5, 8)}/${digits.slice(8)}`;
      return `${digits.slice(0, 2)}.${digits.slice(2, 5)}.${digits.slice(5, 8)}/${digits.slice(8, 12)}-${digits.slice(12, 14)}`;
    }

    /**
     * Remove qualquer caractere não-numérico. Use antes de salvar no banco.
     */
    export function unmask(value: string): string {
      return value.replace(/\D/g, "");
    }
    ```

    import { unmask } from "@/lib/masks";

    /**
     * Valida CPF brasileiro pelo algoritmo de dígitos verificadores.
     * Rejeita os 10 casos de dígitos repetidos (000... a 999...) — passam no
     * algoritmo mas são considerados inválidos pela Receita Federal.
     */
    export function validateCPF(cpf: string): boolean {
      const digits = unmask(cpf);
      if (digits.length !== 11) return false;

      // Rejeita dígitos repetidos (000.000.000-00 etc.)
      if (/^(\d)\1{10}$/.test(digits)) return false;

      const calcDigit = (slice: string, factorStart: number): number => {
        let sum = 0;
        for (let i = 0; i < slice.length; i++) {
          sum += parseInt(slice[i], 10) * (factorStart - i);
        }
        const rest = (sum * 10) % 11;
        return rest === 10 ? 0 : rest;
      };

      const d1 = calcDigit(digits.slice(0, 9), 10);
      if (d1 !== parseInt(digits[9], 10)) return false;

      const d2 = calcDigit(digits.slice(0, 10), 11);
      if (d2 !== parseInt(digits[10], 10)) return false;

      return true;
    }

    /**
     * Valida telefone celular BR: 11 dígitos, DDD entre 11 e 99 (D-10).
     */
    export function validateTelefone(tel: string): boolean {
      const digits = unmask(tel);
      if (digits.length !== 11) return false;
      const ddd = parseInt(digits.slice(0, 2), 10);
      if (ddd < 11 || ddd > 99) return false;
      return true;
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → Supabase Auth | Untrusted email/senha do signup. |
| browser → edge function `create-colaborador` | Body com PII (cpf, telefone, setor) cruza fronteira pública. |
| edge function → public.colaboradores | Service role bypassa RLS; insert com PII. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | edge function `create-colaborador` body | mitigate | Server-side: validar `setor` contra lista fixa antes do insert (DB CHECK pega como segunda barreira). `nome`/`user_id` já obrigatórios. |
| T-02-02 | Information Disclosure | logs do edge function | mitigate | Não logar `cpf` nem `telefone` no `console.log`/`console.error` (PII). Apenas logar `error.message` do Supabase, que não inclui o body. |
| T-02-03 | Tampering | client-side validators (masks/validators.ts) | accept | Defesa em profundidade: cliente valida CPF, mas edge function não re-valida CPF (só setor). Aceito porque CPF inválido não é problema de segurança — é problema de dado. Re-validação seria nice-to-have. |
| T-02-04 | Spoofing | endpoint público de signup | accept | `allowed_users` gate (existing AUTH-03) já filtra emails. Edge function exige `user_id` que veio de `auth.signUp` legítimo. Reuse pattern existente. |
| T-02-05 | Denial of Service | edge function pública | accept | Sem rate limiting customizado. Supabase rate limita por padrão (100 req/min). Marco focado em features, não em hardening — registrar como melhoria futura. |
</threat_model>
